\newcommand{\R}[1]{\mathbb{R}^{#1}}                     % real numbers
\newcommand{\order}[1]{\mathcal{O}\left( #1 \right)}    % order

\newcommand{\p}[1]{\left( #1 \right)}                   % parentheses
\newcommand{\brac}[1]{\left\{ #1 \right\}}              % brackets
\newcommand{\expit}[1]{e^{#1}}                          % exponential
\newcommand{\inv}[1]{#1^{-1}}                           % inverse
\newcommand{\invp}[1]{\inv{\p{#1}}}                     % inverse w/ parentheses

%% Title formatting
\usepackage{titlesec} % to change font settings of titles: http://www.ctex.org/documents/packages/layout/titlesec.pdf, and allow extra subsections using \paragraph{}
\usepackage{titling}
\pretitle{\vspace{-3cm}\begin{flushleft}\Large}
\posttitle{\end{flushleft}\vspace{-0.5cm}}
\preauthor{\begin{flushleft}\large}
\postauthor{\end{flushleft}\vspace{-1.2cm}}
\newcommand{\mytitle}[1]{\title{\textbf{#1}}}
\newcommand{\myauthor}[1]{\author{#1 \hfill \today}\date{}}

%% Section formatting
\titleformat{\section}[block]{\Large\bfseries\filcenter}{\thesection.}{0.4em}{}[\vspace{-0.25em}] % remove \thesection. to remove numbering of sections
\titleformat{\subsection}{\large\bfseries}{\thesubsection}{1em}{}

%% List formatting
\usepackage{enumitem}
\setlist{topsep=5pt,itemsep=5pt,parsep=3pt}
% topsep: space between list and surrounding text
% itemsep: space between list items
% parsep: space between paragraphs w/in a single list item
\setlist[itemize,1]{label=$\triangleright$} % specifies symbol for level 1 of itemize lists

%% Figures
\usepackage{graphicx}
% \graphicspath{{figures_path_1/}{~/figures_path_2/}} # set paths in which to look for figures
\usepackage{float} % to place figures precisely within text using the H option: \begin{figure}[H]
\usepackage{wrapfig} % for text wrapping of figures
\usepackage[font=footnotesize]{caption}  % (cf. http://www.peteryu.ca/tutorials/publishing/latex_captions)
\usepackage[font=scriptsize]{subcaption}
\captionsetup[figure]{labelformat=simple, labelsep=space, skip=0pt} % set labelformat = empty to eliminate figure caption labels
\captionsetup[subfigure]{labelformat=parens, labelsep=space, skip=0pt}
